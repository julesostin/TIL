```
-- SELECT * FROM 테이블이름 WEHRE 조건
-- SELECT age FROM users WHERE >= 30;

-- 30세 이상인 사람들
SELECT * FROM users WHERE age >= 30;

-- 30이상인 유저의 이름만 조회하려면?
SELECT first_name FROM users WHERE age >= 30;

-- user 테이블에서 age 30이상 성이 "김"인 사람의 나이와 이름만
SELECT last_name, age FROM users WHERE age >= 30 AND first_name '김';

-- 비교연산자 / 논리연산자
-- 논리연산자 중 AND / OR / NOT 

-- *주의* 괄호가 없으면 AND앞에있는 값과 AND다음에 있는 조건이 하나여서
-- 괄호를 쳐주자

-- SQL 사용할 수 있는 연산자
-- BETWEEN 값1 AND 값2 : 값1과 값2 사이의 비교 (값1 >= 비교값 >= 값2)
-- IN 목록중에 하나라도 일치하면 성공CC
-- LIKE : 비교 문자열과 형태 일치 / 와일드카드(여러파일을 한꺼번에 지정가능)
-- IS NULL / IS NOT NULL : NULL 여부를 확인할 때는 항상 = 대신에 IS를 사용
-- 부정연산자 : != , ^=, <> / NOT 칼럼명 = (같지않다) / NOT 칼럼명 > (~보다 크지않다)

-- 연산자 우선순위 : 괄호 -> NOT -> SQL,비교연산자 -> AND -> OR

-- SQLite 집계함수
-- 1) COUNT : 항목수를 가져옴
-- 2) AVG : 모든 값의 평균을 계산
-- 3) MAX : 그룹에 있는 모든 값의 최대 치를 가져옴
-- 4) MIN : 그룹에 있는 모든 값의 최소값을 가져옴
-- 5) SUM : 모든 값의 합을 계산

-- COUNT(갯수 조회하기)
-- users 테이블의 레코드 총 갯수를 조회한다면?
SELECT COUNT(*) FROM users;

-- AVG, MIN, MAN, SUM
SELECT AVG(컬럼) FROM 테이블이름;
-- ...다른 함수도 이하 같음 컬럼자리에 MIN,MAX등등 들어가면 됨

-- 30살 이상인 사람들의 평균 나이는?
SELECT AVG(age) FROM users WHERE age>=30;

-- 계좌 잔액이 가장 높은 사람과 그 액수를 조회하려면?
SELECT first_name, MAX(balance) FROM users;

-- 나이가 30살 이상인 사람의 계좌 평균 잔액을 조회?
SELECT AVG(balance) FROM users WHERE age >= 30;

-- 성별(gender)이 1인 사람의 평균 키(height)와 평균 몸무게(weight)
SELECT AVG(weight), AVG(height) FROM healthcare WHERE gender=1; 

-- LIKE
-- 패턴일치를 기반으로 데이터를 조회하는 방법
-- % : 0개 이상의 문자 / _ : 임의의 단일 문자

-- 와일드카드 2가지 패턴
-- % : 이 자리에 문자열이 있을 수도, 없을 수도 있다
-- _ : 반드시 이 자리에 한 개의 문자가 존재해야 한다

SELECT * FROM users WHERE age LIKE '2%';
-- 2로 시작하는 나이는?인듯_

-- 와일드 카드 사용 예시
-- 2% : 2로 시작하는 값.
-- %2 : 2로 끝나는 값
-- %2% : 2가 들어가는 값
-- _2% : 아무 값이 1개(하나)있고 두번째가 2로 시작하는 값
-- 1___ : 1로 시작하고 총 4자리인 값
-- 2_%_% / 2__% : 2로 시작하고 적어도 3자리인 값

-- users 테이블에서 나이가 20대인 사람만 조회하면?
SELECT * FROM users WHERE age LIKE '2_';

-- users 테이블에서 지역 번호가 02인 사람만 조회하면?
SELECT * FROM users WHERE phone LIKE '02-_';

-- users 테이블에서 이름이 '준'으로 끝나는 사람만 조회하면?
SELECT * FROM users WHERE first_name LIKE '%준';

-- users 테이블에서 중간 번호가 5114인 사람만 조회한다면?
SELECT * FROM users WHERE phone LIKE '%-5114-%';

-- ORDER BY
-- SELECT문에 추가하여 사용
-- ASC : 오름차순(디폴트값) / DESC : 내림차순

-- 특정 컬럼을 기준으로 데이터를 정렬해서 조회하기
-- SELECT * FROM 테이블이름 ORDER BY 컬럼 ASC(DESC);

-- users 에서 나이 순으로 오름차순 정렬하여 상위 10개만 조회?
SELECT * FROM users ORDER BY age ASC LIMIT 10;
-- WHERE 들어갔던 자리에 ORDER BY가 들어감

-- 나이 순, 성 순으로 오름차순 상위 10개 조회?
SELECT * FROM users ORDER BY age, last_name ASC LIMIT 10;

-- 계좌 잔액 순으로 내림차순 정렬 유저의 성과 이름 10개만?
SELECT first_name, last_name FROM users ORDER BY balance DESC LIMIT 10;
```
